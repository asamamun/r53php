menu: user accounts > select edit priviledges of root account, change password

go to phpmyadmin folder in xampp. edit config.inc.php change password

Create user

    
CREATE USER user [IDENTIFIED BY [PASSWORD] 'password']
 [, user [IDENTIFIED BY [PASSWORD] 'password']] ...

create user idb identified by 'secret';

create user idb@localhost identified by 'secret';

Granting Privileges

    
GRANT privilege_type [(column_list)] [, privilege_type [(column_list)] ...]
 ON {table_name | * | *.* | database_name.*}
 TO user_name [IDENTIFIED BY 'password']
 [, user_name [IDENTIFIED BY 'password'] ...]
 [REQUIRE {SSL|X509} [ISSUER issuer] [SUBJECT subject]]
 [WITH GRANT OPTION]

grant all privileges on r53_php.* to 'gnsl'@'localhost' identified by 'admin';

grant select, insert on sakila.* to idb@localhost;

Revoking Privileges

    
REVOKE privilege_type [(column_list)] [, privilege_type [(column_list)] 
...]
 ON {table_name | * | *.* | database_name.*}
 FROM user_name [, user_name ...]

revoke all privileges on r53_php.* FROM 'gnsl'@'localhost';

FLUSH PRIVILEGES;

Deleting a User

drop user 'jason'@'localhost';
